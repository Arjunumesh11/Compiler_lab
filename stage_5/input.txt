

decl
  int fact(int n);
enddecl

int fact(int n)
{
  decl
    int f;
  enddecl

  begin
    if(n<=1) then
      f=1;
    else
      f=fact(n-1);
      f=n*f;
    endif;

    return f;
  end
}

int main()
{
  decl
    int res,a;
  enddecl

  begin
    read(a);
    res = fact(a);
    write(res);
    return 0;
  end  
}

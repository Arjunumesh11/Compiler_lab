
	
	class
		fact{
			decl
				int x;
				int findfactorial(int k);
			enddecl
			int findfactorial(int k){
			decl
				int p;
			enddecl
			begin
				if(k<=1)then
				p=1;
				else
				p=k*self.findfactorial(k-1);
				endif;
				return p;
			end
			}
		}
		
	testfactsum{
		decl
			fact o1;
			int testfun(int n);
		enddecl
		int testfun(int n){
			decl
				int sum;
			enddecl
			begin
				self.o1=alloc();
				sum=0;
				while(n!=0)do
					sum=sum+self.o1.findfactorial(n);
					n=n-1;
				endwhile;
				return sum;
			end
			
		}
	}
	endclass

	decl
		testfactsum obj;
	enddecl

	int main(){
		decl
			int x,n;
		enddecl
		begin
			x=initialize();
			obj=alloc();
			read(n);
			write(obj.testfun(n));
			return 1;
		end
